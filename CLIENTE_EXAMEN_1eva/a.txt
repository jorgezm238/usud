import React, { useState } from 'react';

function FiltradoCorredores({ corredores }) {
  const [nacionalidad, setNacionalidad] = useState('');

  // Filtrar los corredores por nacionalidad (sin distinguir mayúsculas/minúsculas)
  const corredoresFiltrados = corredores.filter((corredor) =>
    corredor.nationality.toLowerCase().includes(nacionalidad.toLowerCase())
  );

  return (
    <div>
      <h2>Filtrar Corredores por Nacionalidad</h2>
      <input
        type="text"
        placeholder="Ingrese la nacionalidad"
        value={nacionalidad}
        onChange={(e) => setNacionalidad(e.target.value)}
      />
      <ul>
        {corredoresFiltrados.map((corredor) => (
          <li key={corredor.driverId}>{corredor.driverRef}</li>
        ))}
      </ul>
    </div>
  );
}

export default FiltradoCorredores;
